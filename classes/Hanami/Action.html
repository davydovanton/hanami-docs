def self.included(base)
  base.class_eval do
    include Rack
    include Mime
    include Redirect
    include Exposable
    include Throwable
    include Callbacks
    include Validatable if defined?(Validatable)
    include Configurable
    include Head
    prepend Callable
  end
end
