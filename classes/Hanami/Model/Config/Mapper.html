def initialize(path=nil, &blk)
  if block_given?
    @blk = blk
  elsif path
    @path = root.join(path)
  else
    raise Hanami::Model::InvalidMappingError.new('You must specify a block or a file.')
  end
end

def to_proc
  unless @blk
    code = realpath.read
    @blk = Proc.new { eval(code) }
  end

  @blk
end
