def validations(&blk)
  base   = _build(&_base_rules)
  schema = _build(rules: base.rules, &blk)
  schema.configure(&_schema_predicates)
  schema.configure(&_schema_config)
  schema.extend(__messages) unless _predicates.empty?

  self.schema = schema.new
end
