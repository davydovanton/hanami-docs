def self.coerce(coercer, value, &blk)
  if ::Hanami::Utils::Kernel.respond_to?(coercer.to_s)
    ::Hanami::Utils::Kernel.__send__(coercer.to_s, value, &blk) rescue nil
  else
    coercer.new(value, &blk)
  end
end
