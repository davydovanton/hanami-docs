def validate
  @errors.clear
  @validation_set.each do |name, validations|
    value = @attributes[name]
    value = @attributes[name.to_s] if value.nil?

    attribute = Attribute.new(@attributes, name, value, validations, @errors)
    attribute.validate
  end
  @errors
end
