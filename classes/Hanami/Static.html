def call(env)
  path           = env[PATH_INFO]

  prefix, config = @sources.find { |p, _| path.start_with?(p) }
  if prefix && config
    original = config.sources.find(path.sub(prefix, ''))
  end

  if can_serve(path, original)
    super
  else
    precompile(original, config) ?
      call(env) :
      @app.call(env)
  end
end
