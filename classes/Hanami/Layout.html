def self.included(base)
  conf = Hanami::View::Configuration.for(base)
  conf.add_layout(base)

  base.class_eval do
    extend Hanami::View::Dsl.dup
    extend ClassMethods

    include Utils::ClassAttribute
    class_attribute :configuration

    self.configuration = conf.duplicate
  end

  conf.copy!(base)
end
