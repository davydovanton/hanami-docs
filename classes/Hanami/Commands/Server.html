def start
  case @strategy
  when :entr
    exec ENTR_EXECUTE_COMMAND % {paths: project_paths, args: server_options}
  when :shotgun
    Shotgun.enable_copy_on_write
    Shotgun.preload
    @server.start
  when :rackup
    @server.start
  end
end
