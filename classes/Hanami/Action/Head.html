def finish
  super

  if _requires_no_body?
    @_body = nil
    @headers.reject! {|header,_| !keep_response_header?(header) }
  end
end

def _requires_no_body?
  HTTP_STATUSES_WITHOUT_BODY.include?(@_status) || head?
end
