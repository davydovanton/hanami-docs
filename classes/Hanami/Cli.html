def new(application_name)
  if options[:help]
    invoke :help, ['new']
  elsif options[:architecture] == 'app'
    Hanami::Commands::New::App.new(options, application_name).start
  else
    Hanami::Commands::New::Container.new(options, application_name).start
  end
end
